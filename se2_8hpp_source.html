388  inline static
389  const SE2Group<Scalar> exp(const Tangent & a) {
390  Scalar theta = a[2];
391  const SO2Group<Scalar> & so2 = SO2Group<Scalar>::exp(theta);
392  Scalar sin_theta_by_theta;
393  Scalar one_minus_cos_theta_by_theta;
394 
395  if(std::abs(theta)<SophusConstants<Scalar>::epsilon()) {
396  Scalar theta_sq = theta*theta;
397  sin_theta_by_theta
398  = static_cast<Scalar>(1.) - static_cast<Scalar>(1./6.)*theta_sq;
399  one_minus_cos_theta_by_theta
400  = static_cast<Scalar>(0.5)*theta
401  - static_cast<Scalar>(1./24.)*theta*theta_sq;
402  } else {
403  sin_theta_by_theta = so2.unit_complex().y()/theta;
404  one_minus_cos_theta_by_theta
405  = (static_cast<Scalar>(1.) - so2.unit_complex().x())/theta;
406  }
407  Matrix<Scalar,2,1> trans
408  (sin_theta_by_theta*a[0] - one_minus_cos_theta_by_theta*a[1],
409  one_minus_cos_theta_by_theta * a[0]+sin_theta_by_theta*a[1]);
410  return SE2Group<Scalar>(so2, trans);
411  }

489  inline static
490  const Tangent lieBracket(const Tangent & a,
491  const Tangent & b) {
492  Matrix<Scalar,2,1> upsilon1 = a.template head<2>();
493  Matrix<Scalar,2,1> upsilon2 = b.template head<2>();
494  Scalar theta1 = a[2];

518  inline static
519  const Tangent log(const SE2Group<Scalar> & other) {
520  Tangent upsilon_theta;
521  const SO2Group<Scalar> & so2 = other.so2();
522  Scalar theta = SO2Group<Scalar>::log(so2);
523  upsilon_theta[2] = theta;
524  Scalar halftheta = static_cast<Scalar>(0.5)*theta;
525  Scalar halftheta_by_tan_of_halftheta;
526 
527  const Matrix<Scalar,2,1> & z = so2.unit_complex();
528  Scalar real_minus_one = z.x()-static_cast<Scalar>(1.);
529  if (std::abs(real_minus_one)<SophusConstants<Scalar>::epsilon()) {
530  halftheta_by_tan_of_halftheta
531  = static_cast<Scalar>(1.)
532  - static_cast<Scalar>(1./12)*theta*theta;
533  } else {
534  halftheta_by_tan_of_halftheta
535  = -(halftheta*z.y())/(real_minus_one);
536  }
537  Matrix<Scalar,2,2> V_inv;
538  V_inv << halftheta_by_tan_of_halftheta, halftheta
539  , -halftheta, halftheta_by_tan_of_halftheta;
540  upsilon_theta.template head<2>() = V_inv*other.translation();
541  return upsilon_theta;
542  }

554  inline static

556  Tangent upsilon_omega;
557  upsilon_omega.template head<2>() = Omega.col(2).template head<2>();
558  upsilon_omega[2]
559  = SO2Group<Scalar>::vee(Omega.template topLeftCorner<2,2>());
