406  inline static
407  const SE3Group<Scalar> exp(const Tangent & a) {
408  const Matrix<Scalar,3,1> & omega = a.template tail<3>();
409 
410  Scalar theta;
411  const SO3Group<Scalar> & so3
412  = SO3Group<Scalar>::expAndTheta(omega, &theta);
413 
414  const Matrix<Scalar,3,3> & Omega = SO3Group<Scalar>::hat(omega);
415  const Matrix<Scalar,3,3> & Omega_sq = Omega*Omega;
416  Matrix<Scalar,3,3> V;
417 
418  if(theta<SophusConstants<Scalar>::epsilon()) {
419  V = so3.matrix();
420  //Note: That is an accurate expansion!
421  } else {
422  Scalar theta_sq = theta*theta;
423  V = (Matrix<Scalar,3,3>::Identity()
424  + (static_cast<Scalar>(1)-std::cos(theta))/(theta_sq)*Omega
425  + (theta-std::sin(theta))/(theta_sq*theta)*Omega_sq);
426  }
427  return SE3Group<Scalar>(so3,V*a.template head<3>());
428  }

528  inline static
529  const Tangent lieBracket(const Tangent & a,
530  const Tangent & b) {
531  Matrix<Scalar,3,1> upsilon1 = a.template head<3>();
532  Matrix<Scalar,3,1> upsilon2 = b.template head<3>();
533  Matrix<Scalar,3,1> omega1 = a.template tail<3>();
534  Matrix<Scalar,3,1> omega2 = b.template tail<3>();
535 
536  Tangent res;
537  res.template head<3>() = omega1.cross(upsilon2) + upsilon1.cross(omega2);
538  res.template tail<3>() = omega1.cross(omega2);
539 
540  return res;
541  }

559  inline static
560  const Tangent log(const SE3Group<Scalar> & se3) {
561  Tangent upsilon_omega;
562  Scalar theta;
563  upsilon_omega.template tail<3>()
564  = SO3Group<Scalar>::logAndTheta(se3.so3(), &theta);
565 
566  if (std::abs(theta)<SophusConstants<Scalar>::epsilon()) {
567  const Matrix<Scalar,3,3> & Omega
568  = SO3Group<Scalar>::hat(upsilon_omega.template tail<3>());
569  const Matrix<Scalar,3,3> & V_inv =
570  Matrix<Scalar,3,3>::Identity() -
571  static_cast<Scalar>(0.5)*Omega
572  + static_cast<Scalar>(1./12.)*(Omega*Omega);
573 
574  upsilon_omega.template head<3>() = V_inv*se3.translation();
575  } else {
576  const Matrix<Scalar,3,3> & Omega
577  = SO3Group<Scalar>::hat(upsilon_omega.template tail<3>());
578  const Matrix<Scalar,3,3> & V_inv =
579  ( Matrix<Scalar,3,3>::Identity() - static_cast<Scalar>(0.5)*Omega
580  + ( static_cast<Scalar>(1)
581  - theta/(static_cast<Scalar>(2)*tan(theta/Scalar(2)))) /
582  (theta*theta)*(Omega*Omega) );
583  upsilon_omega.template head<3>() = V_inv*se3.translation();
584  }
585  return upsilon_omega;
586  }

598  inline static

600  Tangent upsilon_omega;
601  upsilon_omega.template head<3>() = Omega.col(3).template head<3>();
602  upsilon_omega.template tail<3>()
