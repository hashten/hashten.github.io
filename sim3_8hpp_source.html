130  const Adjoint Adj() const {
131  const Matrix<Scalar,3,3> & R = rxso3().rotationMatrix();
132  Adjoint res;
133  res.setZero();
134  res.block(0,0,3,3) = scale()*R;

136  res.block(0,6,3,1) = -translation();

384  inline static
385  const Adjoint d_lieBracketab_by_d_a(const Tangent & b) {
386  const Matrix<Scalar,3,1> & upsilon2 = b.template head<3>();
387  const Matrix<Scalar,3,1> & omega2 = b.template segment<3>(3);
388  Scalar sigma2 = b[6];
389 
390  Adjoint res;
391  res.setZero();
392  res.template topLeftCorner<3,3>()
393  = -SO3::hat(omega2)-sigma2*Matrix3d::Identity();
394  res.template block<3,3>(0,3) = -SO3::hat(upsilon2);
395  res.template topRightCorner<3,1>() = upsilon2;
396  res.template block<3,3>(3,3) = -SO3::hat(omega2);
397  return res;
398  }

417  inline static
418  const Sim3Group<Scalar> exp(const Tangent & a) {
419  const Matrix<Scalar,3,1> & upsilon = a.segment(0,3);
420  const Matrix<Scalar,3,1> & omega = a.segment(3,3);
421  Scalar sigma = a[6];
422  Scalar theta;
423  RxSO3Group<Scalar> rxso3
424  = RxSO3Group<Scalar>::expAndTheta(a.template tail<4>(), &theta);
425  const Matrix<Scalar,3,3> & Omega = SO3Group<Scalar>::hat(omega);
426  const Matrix<Scalar,3,3> & W = calcW(theta, sigma, rxso3.scale(), Omega);
427  return Sim3Group<Scalar>(rxso3, W*upsilon);
428  }

534  inline static
535  const Tangent lieBracket(const Tangent & a,
536  const Tangent & b) {
537  const Matrix<Scalar,3,1> & upsilon1 = a.template head<3>();
538  const Matrix<Scalar,3,1> & upsilon2 = b.template head<3>();
539  const Matrix<Scalar,3,1> & omega1 = a.template segment<3>(3);
540  const Matrix<Scalar,3,1> & omega2 = b.template segment<3>(3);
541  Scalar sigma1 = a[6];
542  Scalar sigma2 = b[6];
543 
544  Tangent res;
545  res.template head<3>() =
546  SO3Group<Scalar>::hat(omega1)*upsilon2
547  + SO3Group<Scalar>::hat(upsilon1)*omega2
548  + sigma1*upsilon2 - sigma2*upsilon1;
549  res.template segment<3>(3) = omega1.cross(omega2);
550  res[6] = static_cast<Scalar>(0);
551 
552  return res;
553  }

571  inline static
572  const Tangent log(const Sim3Group<Scalar> & other) {
573  Tangent res;
574  Scalar theta;
575  const Matrix<Scalar,4,1> & omega_sigma
576  = RxSO3Group<Scalar>::logAndTheta(other.rxso3(), &theta);
577  const Matrix<Scalar,3,1> & omega = omega_sigma.template head<3>();
578  Scalar sigma = omega_sigma[3];
579  const Matrix<Scalar,3,3> & W
580  = calcW(theta, sigma, other.scale(), SO3Group<Scalar>::hat(omega));
581  res.segment(0,3) = W.partialPivLu().solve(other.translation());
582  res.segment(3,3) = omega;
583  res[6] = sigma;
584  return res;
585  }

597  inline static
598  const Tangent vee(const Transformation & Omega) {
599  Tangent upsilon_omega_sigma;
600  upsilon_omega_sigma.template head<3>()
601  = Omega.col(3).template head<3>();
602  upsilon_omega_sigma.template tail<4>()

604  return upsilon_omega_sigma;

607 private:
608  static
609  Matrix<Scalar,3,3> calcW(const Scalar & theta,
610  const Scalar & sigma,
611  const Scalar & scale,
612  const Matrix<Scalar,3,3> & Omega){
613  static const Matrix<Scalar,3,3> I
614  = Matrix<Scalar,3,3>::Identity();
615  static const Scalar one = static_cast<Scalar>(1.);
616  static const Scalar half = static_cast<Scalar>(1./2.);
617  Matrix<Scalar,3,3> Omega2 = Omega*Omega;
618 
619  Scalar A,B,C;
620  if (std::abs(sigma)<SophusConstants<Scalar>::epsilon()) {
621  C = one;
622  if (std::abs(theta)<SophusConstants<Scalar>::epsilon()) {
623  A = half;
624  B = static_cast<Scalar>(1./6.);
625  } else {
626  Scalar theta_sq = theta*theta;
627  A = (one-std::cos(theta))/theta_sq;
628  B = (theta-std::sin(theta))/(theta_sq*theta);
629  }
630  } else {
631  C = (scale-one)/sigma;
632  if (std::abs(theta)<SophusConstants<Scalar>::epsilon()) {
633  Scalar sigma_sq = sigma*sigma;
634  A = ((sigma-one)*scale+one)/sigma_sq;
635  B = ((half*sigma*sigma-sigma+one)*scale)/(sigma_sq*sigma);
636  } else {
637  Scalar theta_sq = theta*theta;
638  Scalar a = scale*std::sin(theta);
639  Scalar b = scale*std::cos(theta);
640  Scalar c = theta_sq+sigma*sigma;
641  A = (a*sigma+ (one-b)*theta)/(theta*c);
642  B = (C-((b-one)*sigma+a*theta)/(c))*one/(theta_sq);
643  }
644  }
645  return A*Omega + B*Omega2 + C*I;
646  }
